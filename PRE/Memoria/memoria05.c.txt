#include <stdio.h>
#include <stdlib.h>

// Função para criar e ler o vetor dinamicamente
int *criarVetor(int n)
{
    int *vetor;
    int i;

    vetor = (int *) malloc(n * sizeof(int));

    if(vetor == NULL)
    {
        printf("Erro ao alocar memoria.\n");
        return NULL;
    }

    printf("Digite os valores do vetor:\n");

    for(i = 0; i < n; i++)
    {
        printf("Valor %d: ", i + 1);
        scanf("%d", &vetor[i]);
    }

    return vetor;
}

// Função para contar e mostrar os múltiplos de X
void mostrarMultiplos(int *vetor, int n, int x)
{
    int i;
    int contador = 0;

    printf("\nMultiplos de %d encontrados:\n", x);

    for(i = 0; i < n; i++)
    {
        if(vetor[i] % x == 0)
        {
            printf("%d\n", vetor[i]);
            contador++;
        }
    }

    printf("Quantidade de multiplos: %d\n", contador);
}

int main()
{
    int n, x;
    int *vetor;

    printf("Digite o tamanho do vetor: ");
    scanf("%d", &n);

    vetor = criarVetor(n);

    if(vetor == NULL)
        return 1;

    printf("\nDigite o numero X: ");
    scanf("%d", &x);

    mostrarMultiplos(vetor, n, x);

    free(vetor);

    return 0;
}
