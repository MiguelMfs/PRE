#include <stdio.h>
#include <stdlib.h>

// Função para ler os valores do vetor
void lerVetor(int *vetor, int tamanho)
{
    int i;

    for(i = 0; i < tamanho; i++)
    {
        printf("Digite o valor %d: ", i + 1);
        scanf("%d", (vetor + i));
    }
}

// Função para contar pares e ímpares
void contarParesImpares(int *vetor, int tamanho, int *pares, int *impares)
{
    int i;

    *pares = 0;
    *impares = 0;

    for(i = 0; i < tamanho; i++)
    {
        if(*(vetor + i) % 2 == 0)
            (*pares)++;
        else
            (*impares)++;
    }
}

int main()
{
    int *vetor;
    int tamanho;
    int pares, impares;

    printf("Digite o tamanho do vetor: ");
    scanf("%d", &tamanho);

    vetor = (int *) malloc(tamanho * sizeof(int));

    if(vetor == NULL)
    {
        printf("Erro ao alocar memoria.\n");
        return 1;
    }

    lerVetor(vetor, tamanho);

    contarParesImpares(vetor, tamanho, &pares, &impares);

    printf("\nQuantidade de numeros pares: %d\n", pares);
    printf("Quantidade de numeros impares: %d\n", impares);

    free(vetor);

    return 0;
}
