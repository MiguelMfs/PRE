#include <stdio.h>
#include <stdlib.h>

int main()
{
    int *vetor;
    int tamanho;
    int i;

    // Leitura do tamanho do vetor
    printf("Digite o tamanho do vetor: ");
    scanf("%d", &tamanho);

    // Alocação dinâmica
    vetor = (int *) malloc(tamanho * sizeof(int));

    if(vetor == NULL)
    {
        printf("Erro ao alocar memoria.\n");
        return 1;
    }

    // Leitura dos valores
    printf("Digite os valores do vetor:\n");

    for(i = 0; i < tamanho; i++)
    {
        printf("Valor %d: ", i + 1);
        scanf("%d", &vetor[i]);
    }

    // Impressão do vetor
    printf("\nVetor lido:\n");

    for(i = 0; i < tamanho; i++)
    {
        printf("%d ", vetor[i]);
    }

    printf("\n");

    // Liberação da memória
    free(vetor);

    return 0;
}
