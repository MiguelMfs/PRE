#include <stdio.h>
#include <stdlib.h>

int main()
{
    int *vetor;
    int valor;
    int quantidade = 0;
    int i;

    vetor = NULL;

    printf("Digite numeros inteiros (numero negativo encerra):\n");

    while(1)
    {
        printf("Digite um valor: ");
        scanf("%d", &valor);

        if(valor < 0)
            break;

        // Realoca espaço para mais um elemento
        vetor = (int *) realloc(vetor, (quantidade + 1) * sizeof(int));

        if(vetor == NULL)
        {
            printf("Erro ao alocar memoria.\n");
            return 1;
        }

        vetor[quantidade] = valor;
        quantidade++;
    }

    // Impressão do vetor lido
    printf("\nVetor lido:\n");

    for(i = 0; i < quantidade; i++)
    {
        printf("%d ", vetor[i]);
    }

    printf("\n");

    // Liberação da memória
    free(vetor);

    return 0;
}
