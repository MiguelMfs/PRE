#include <stdio.h>

#define V 5
#define A 5

typedef struct
{
    int codigo;
    int voosSaem;
    int voosChegam;
} Aeroporto;

typedef struct
{
    int origem;
    int destino;
} Voo;

int main()
{
    Aeroporto aeroportos[A];
    Voo voos[V];
    int i;

    // Inicializa os aeroportos
    for(i = 0; i < A; i++)
    {
        aeroportos[i].codigo = i;
        aeroportos[i].voosSaem = 0;
        aeroportos[i].voosChegam = 0;
    }

    // Leitura dos voos
    for(i = 0; i < V; i++)
    {
        do
        {
            printf("\nVoo %d\n", i + 1);
            printf("Codigo do aeroporto de origem (0 a %d): ", A - 1);
            scanf("%d", &voos[i].origem);
        } while(voos[i].origem < 0 || voos[i].origem >= A);

        do
        {
            printf("Codigo do aeroporto de destino (0 a %d): ", A - 1);
            scanf("%d", &voos[i].destino);
        } while(voos[i].destino < 0 || voos[i].destino >= A);

        aeroportos[voos[i].origem].voosSaem++;
        aeroportos[voos[i].destino].voosChegam++;
    }

    // Listagem dos aeroportos e seus voos
    printf("\n===== AEROPORTOS =====\n");

    for(i = 0; i < A; i++)
    {
        printf("\nAeroporto %d\n", aeroportos[i].codigo);
        printf("Voos que saem: %d\n", aeroportos[i].voosSaem);
        printf("Voos que chegam: %d\n", aeroportos[i].voosChegam);

        printf("Voos:\n");

        for(int j = 0; j < V; j++)
        {
            if(voos[j].origem == i || voos[j].destino == i)
            {
                printf("Origem: %d -> Destino: %d\n",
                       voos[j].origem,
                       voos[j].destino);
            }
        }
    }

    return 0;
}
