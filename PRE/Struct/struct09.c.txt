#include <stdio.h>

typedef struct
{
    char nome[100];
    int matricula;
    float mediaFinal;
} Aluno;

int main()
{
    Aluno alunos[10];
    Aluno aprovados[10];
    Aluno reprovados[10];

    int i;
    int qtdAprovados = 0;
    int qtdReprovados = 0;

    // Leitura dos dados
    for (i = 0; i < 10; i++)
    {
        printf("\nAluno %d\n", i + 1);

        printf("Nome: ");
        fgets(alunos[i].nome, sizeof(alunos[i].nome), stdin);

        printf("Matricula: ");
        scanf("%d", &alunos[i].matricula);

        printf("Media Final: ");
        scanf("%f", &alunos[i].mediaFinal);
        getchar();

        // Separação dos alunos
        if (alunos[i].mediaFinal >= 5.0)
        {
            aprovados[qtdAprovados] = alunos[i];
            qtdAprovados++;
        }
        else
        {
            reprovados[qtdReprovados] = alunos[i];
            qtdReprovados++;
        }
    }

    // Exibição dos aprovados
    printf("\n========== APROVADOS ==========\n");

    for (i = 0; i < qtdAprovados; i++)
    {
        printf("\nNome: %s", aprovados[i].nome);
        printf("Matricula: %d\n", aprovados[i].matricula);
        printf("Media Final: %.2f\n", aprovados[i].mediaFinal);
    }

    // Exibição dos reprovados
    printf("\n========== REPROVADOS ==========\n");

    for (i = 0; i < qtdReprovados; i++)
    {
        printf("\nNome: %s", reprovados[i].nome);
        printf("Matricula: %d\n", reprovados[i].matricula);
        printf("Media Final: %.2f\n", reprovados[i].mediaFinal);
    }

    return 0;
}
