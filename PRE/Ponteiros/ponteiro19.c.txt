#include <stdio.h>

void encontrarMinMax(int *v, int tamanho, int *min, int *max)
{
    int i;

    *min = v[0];
    *max = v[0];

    for(i = 1; i < tamanho; i++)
    {
        if(*(v + i) < *min)
            *min = *(v + i);

        if(*(v + i) > *max)
            *max = *(v + i);
    }
}

int main()
{
    int vetor[5];
    int minimo, maximo;
    int i;

    printf("Digite 5 valores inteiros:\n");

    for(i = 0; i < 5; i++)
    {
        scanf("%d", &vetor[i]);
    }

    encontrarMinMax(vetor, 5, &minimo, &maximo);

    printf("\nMenor valor: %d\n", minimo);
    printf("Maior valor: %d\n", maximo);

    return 0;
}
