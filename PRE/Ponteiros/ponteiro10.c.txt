#include <stdio.h>

int main()
{
    int vetor[5];
    int *p;
    int i;

    p = vetor;

    // Leitura do array usando apenas aritmética de ponteiros
    for(i = 0; i < 5; i++)
    {
        printf("Digite o valor %d: ", i + 1);
        scanf("%d", p + i);
    }

    // Impressão do dobro dos valores usando ponteiros
    printf("\nDobro dos valores:\n");

    for(i = 0; i < 5; i++)
    {
        printf("%d\n", (*(p + i)) * 2);
    }

    return 0;
}
