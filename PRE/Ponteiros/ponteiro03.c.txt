#include <stdio.h>

int main()
{
    int a, b;

    printf("Digite o valor da primeira variavel: ");
    scanf("%d", &a);

    printf("Digite o valor da segunda variavel: ");
    scanf("%d", &b);

    if (&a > &b)
        printf("Conteudo do maior endereco: %d\n", a);
    else
        printf("Conteudo do maior endereco: %d\n", b);

    return 0;
}
