#include <stdio.h>

// Função que verifica se a segunda string está dentro da primeira
int verificarSubstring(char *str1, char *str2)
{
    char *p1, *p2;

    while (*str1 != '\0')
    {
        p1 = str1;
        p2 = str2;

        while (*p1 != '\0' && *p2 != '\0' && *p1 == *p2)
        {
            p1++;
            p2++;
        }

        if (*p2 == '\0')
            return 1; // encontrou a segunda string dentro da primeira

        str1++;
    }

    return 0; // não encontrou
}

int main()
{
    char primeira[100];
    char segunda[50];

    printf("Digite a primeira string: ");
    fgets(primeira, sizeof(primeira), stdin);

    printf("Digite a segunda string: ");
    fgets(segunda, sizeof(segunda), stdin);

    // Remove quebra de linha
    char *p;
    p = primeira;
    while (*p != '\0')
    {
        if (*p == '\n')
            *p = '\0';
        p++;
    }

    p = segunda;
    while (*p != '\0')
    {
        if (*p == '\n')
            *p = '\0';
        p++;
    }

    if (verificarSubstring(primeira, segunda))
        printf("A segunda string ocorre dentro da primeira.\n");
    else
        printf("A segunda string nao ocorre dentro da primeira.\n");

    return 0;
}
