#include <stdio.h>

void preencher(int *array, int tamanho, int valor)
{
    int *p;

    p = array;

    while (p < array + tamanho)
    {
        *p = valor;
        p++;
    }
}

int main()
{
    int vetor[10];
    int i;

    preencher(vetor, 10, 5);

    printf("Array preenchido:\n");

    for(i = 0; i < 10; i++)
    {
        printf("%d ", vetor[i]);
    }

    return 0;
}
