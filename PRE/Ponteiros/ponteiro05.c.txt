#include <stdio.h>

// Função que organiza os valores: maior na primeira variável e menor na segunda
void ordenar(int *a, int *b)
{
    int aux;

    if (*a < *b)
    {
        aux = *a;
        *a = *b;
        *b = aux;
    }
}

int main()
{
    int valor1, valor2;

    printf("Digite o primeiro valor: ");
    scanf("%d", &valor1);

    printf("Digite o segundo valor: ");
    scanf("%d", &valor2);

    ordenar(&valor1, &valor2);

    printf("\nMaior valor: %d\n", valor1);
    printf("Menor valor: %d\n", valor2);

    return 0;
}
